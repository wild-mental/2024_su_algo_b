## 문제풀이 중점사항 🤔
```
const fs = require('fs');
const input = fs.readFileSync("/dev/stdin").toString().trim()
    .split('\n').map(Number)
const a = input[0];
const b = input[1];
const arr = input.slice(2);

const result = (a, b, arr) => {
    if(a === 1){
        return 0;  
    } 

    let max = Math.max(...arr);
    let count = 0;

    while(b <= max){
        arr[arr.indexOf(max)] -= 1
        b += 1;
        count++;
        max = Math.max(...arr);
    }

    return count;
}

console.log(result(a, b, arr));
```
